        // 圆形粒子贴图

        function createCircleTexture() {

            const canvas = document.createElement('canvas');

            canvas.width = 128; canvas.height = 128;

            const ctx = canvas.getContext('2d');

            const grad = ctx.createRadialGradient(64, 64, 0, 64, 64, 64);

            grad.addColorStop(0, 'rgba(255, 255, 255, 1)');

            grad.addColorStop(0.3, 'rgba(0, 180, 255, 0.7)');

            grad.addColorStop(0.6, 'rgba(0, 50, 255, 0.1)');

            grad.addColorStop(1, 'rgba(0, 0, 0, 0)');

            ctx.fillStyle = grad;

            ctx.fillRect(0, 0, 128, 128);

            return new THREE.CanvasTexture(canvas);

        }


        // 不规则行星几何体

        function createAsteroidGeometry(radius) {

            const geo = new THREE.IcosahedronGeometry(radius, 1);

            const pos = geo.getAttribute('position');

            for (let i = 0; i < pos.count; i++) {

                const x = pos.getX(i), y = pos.getY(i), z = pos.getZ(i);

                const noise = 0.85 + Math.random() * 0.35;

                pos.setXYZ(i, x * noise, y * noise, z * noise);

            }

            geo.computeVertexNormals();

            return geo;

        }


        function initScene() {

            scene = new THREE.Scene();

            scene.fog = new THREE.FogExp2(0x000205, 0.04);


            camera = new THREE.PerspectiveCamera(70, window.innerWidth / window.innerHeight, 0.1, 1000);

            camera.position.z = 13;


            renderer = new THREE.WebGLRenderer({ antialias: true });

            renderer.setSize(window.innerWidth, window.innerHeight);

            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

            document.body.appendChild(renderer.domElement);


            // 1. 更加密集的粒子宇宙

            const particleCount = 15000;

            const posAttr = new Float32Array(particleCount * 3);

            for (let i = 0; i < particleCount; i++) {

                posAttr[i * 3] = (Math.random() - 0.5) * 70;

                posAttr[i * 3 + 1] = (Math.random() - 0.5) * 70;

                posAttr[i * 3 + 2] = (Math.random() - 0.5) * 70;

            }

            const partGeo = new THREE.BufferGeometry();

            partGeo.setAttribute('position', new THREE.BufferAttribute(posAttr, 3));

            const partMat = new THREE.PointsMaterial({

                size: 0.15, map: createCircleTexture(), transparent: true, opacity: 0.7,

                blending: THREE.AdditiveBlending, depthWrite: false

            });

            particles = new THREE.Points(partGeo, partMat);

            scene.add(particles);


            // 2. 发光小行星

            const asteroidMat = new THREE.MeshPhongMaterial({ 

                color: 0x00aaff, 

                emissive: 0x005577, // 发光底色

                emissiveIntensity: 1,

                shininess: 50,

                flatShading: true

            });

            

            scene.add(new THREE.AmbientLight(0x204060, 1));

            const mainLight = new THREE.DirectionalLight(0xffffff, 1.5);

            mainLight.position.set(5, 10, 10);

            scene.add(mainLight);


            const linePoints = [];

        function createTextTexture(text) {

            const canvas = document.createElement('canvas');

            const ctx = canvas.getContext('2d');

            canvas.width = 1024; canvas.height = 512;

            ctx.fillStyle = '#ffffff';

            ctx.font = 'bold 100px "Microsoft YaHei"'; // 增大字号

            ctx.textAlign = 'center';

            ctx.shadowBlur = 30;

            ctx.shadowColor = '#00f2ff';

            ctx.fillText(text, 512, 256);

            return new THREE.CanvasTexture(canvas);

        }

        const hands = new Hands({

            modelComplexity: 1,

            minDetectionConfidence: 0.8,

            minTrackingConfidence: 0.8

        hands.onResults(onResults);

        function onResults(results) {

            const canvasEle = document.getElementById('gesture-canvas');

            canvasCtx.save();

            canvasCtx.clearRect(0, 0, canvasEle.width, canvasEle.height);

            canvasCtx.drawImage(results.image, 0, 0, canvasEle.width, canvasEle.height);


            if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {

                const landmarks = results.multiHandLandmarks[0];

                drawConnectors(canvasCtx, landmarks, HAND_CONNECTIONS, { color: '#00f2ff', lineWidth: 4 });

                drawLandmarks(canvasCtx, landmarks, { color: '#ffffff', lineWidth: 1, radius: 4 });


                // 准星追踪 (食指尖)

                const point = landmarks[8];

                mouse.x = (1 - point.x) * 2 - 1;

                mouse.y = (1 - point.y) * 2 - 1;


                // 更新 3D 准星

                const vector = new THREE.Vector3(mouse.x, mouse.y, 0.5).unproject(camera);

                const dir = vector.sub(camera.position).normalize();

                const dist = -camera.position.z / dir.z;

                cursor.position.copy(camera.position.clone().add(dir.multiplyScalar(dist)));


                // 手势计算：握拳判定

                const wrist = landmarks[0];

                const tips = [8, 12, 16, 20];

                let totalDist = 0;

                tips.forEach(id => {

                    const t = landmarks[id];

                    totalDist += Math.sqrt(Math.pow(t.x - wrist.x, 2) + Math.pow(t.y - wrist.y, 2));

                });

                const avgDist = totalDist / tips.length;


                // 交互逻辑：avgDist < 0.28 视为握拳抓取

                const isGrabbingNow = avgDist < 0.28;


                if (isGrabbingNow) {

                    if (!isGrabbing) {

                        isGrabbing = true;

                        checkInteraction();

                    }

                } else {

                    if (isGrabbing) {

                        isGrabbing = false;

                        hideContent();

                    }

                }


                cursor.material.color.set(isGrabbing ? 0x00ff00 : 0xffffff);

                cursor.scale.setScalar(isGrabbing ? 0.8 : 1.2);

            }

            canvasCtx.restore();

        }


        function checkInteraction() {

            raycaster.setFromCamera(mouse, camera);

            const intersects = raycaster.intersectObjects(hitboxes);

            if (intersects.length > 0) {

                showContent(intersects[0].object.userData.index);

            }

        }


        function showContent(index) {

            const data = storyData[index];

            document.getElementById('content-title').innerText = data.title;

            document.getElementById('content-body').innerText = data.body;

            document.getElementById('content-overlay').classList.add('visible');

            activeNodeIndex = index;

            

            nodes.forEach((n, i) => {

                const isActive = i === index;

                n.material.emissive.set(isActive ? 0x00ffff : 0x005577);

                n.material.emissiveIntensity = isActive ? 2.5 : 1;

                n.scale.setScalar(isActive ? 1.8 : 1);

                nodeLights[i].intensity = isActive ? 15 : 0;

            });

        }


        function hideContent() {

            document.getElementById('content-overlay').classList.remove('visible');

            nodes.forEach((n, i) => {

                n.material.emissive.set(0x005577);

                n.material.emissiveIntensity = 1;

                n.scale.setScalar(1);

                nodeLights[i].intensity = 0;

            });

            activeNodeIndex = -1;

        }


        function animate() {

            requestAnimationFrame(animate);


            particles.rotation.y += 0.0003;

            particles.rotation.x += 0.0001;


            const time = Date.now() * 0.001;

            

            nodes.forEach((node, i) => {

                node.rotation.y += 0.02;

                node.position.y += Math.sin(time + i) * 0.005;

                

                if (i === activeNodeIndex) {

                    // 激活时的呼吸发光

                    node.material.emissiveIntensity = 2 + Math.sin(time * 8) * 1;

                }

            });


            // 连线跟随行星微动

            const posArr = lines.geometry.attributes.position.array;

            nodes.forEach((node, i) => {

                posArr[i * 3] = node.position.x;

                posArr[i * 3 + 1] = node.position.y;

                posArr[i * 3 + 2] = node.position.z;

            });

            lines.geometry.attributes.position.needsUpdate = true;


            renderer.render(scene, camera);

        }
